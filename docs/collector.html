    /**

    • @function CS
    • @description this is our factory for binding any itteratable functions to.
    • @arg {array|object} */
        function CS(collection) {
      var i = 0,
          len = collection.length;
      for (; i < len; i++) {
        this[i] = collection[i];
      }
      this.length = len;
      this.splice = Array.prototype.splice;
      this.each = Array.prototype.forEach;
      this.indexOf = Array.prototype.indexOf;
      this.some = Array.prototype.some;
    }

    • memberof class api
    • arg string :: class name to be added to all nodes in collection.
    • returns modified collection

    example

    <div class="example">Hello World</div>

     $('.example').addClass('test');

    (function ($) {
  $.plugin('addClass', function (klass) {
    this.each(function (node) {
      if ($(node).hasClass(klass) === false) {
        node.className += ' ' + klass;
      };
    });

    return this;
  });
})(collector);

    hasClass

    hasClass should return true if any of the nodes contain the class you sent in

    • memberof class api
    • arg string :: class name to be checked across all nodes in collection.

    • returns boolean

      example

      <div class="test hasclass">Hello World</div>

      $('.test').hasClass('hasClass');
// true

    (function ($) {
  $.plugin('hasClass', function (klass) {
    return this.some(function (node) {
      var cn = node.className,
          reg = new RegExp(klass);

      return reg.test(cn);
    });
  });
})(collector);

    removeClass

    removeClass should take collections and remove the passed class name on any node that has it

    • memberof class api
    • arg string :: class name to be removed across all nodes in collection.
    • returns modified collection

    example

    <div class="example test">Hello World</div>

    $('.example').removeClass('test')

    <div class="example">Hello World</div>

    (function ($) {
  $.plugin('removeClass', function (klass) {
    this.each(function (node) {
      node.className = node.className.replace(klass, '');
    });

    return this;
  });
})(collector);
